    <script defer src="https://unpkg.com/browse/@slidy/core@<version>/dist/slidy.js"></script>

<script>
    let slidy, node

    window.onload = () => {
        node = document.querySelector('#slidy');
        slidy = Slidy.slidy(node, {
                    index: 0,
                    axis: 'x',
                    align: 'middle',
                    duration: 375,
                    snap: true,
                    gravity: 1.2,
                    clamp: false,
                    loop: false,
                    indexer: (x) => indexing(x),
                    scroller: (p) => scrolling(p),
                });
        slidy.to(5);
    }
</script>
